# دليل بناء نظام OTP / Forgot Password / Reset Password
## مرجع معماري كامل — مشروع ExpoCore (Flutter + GetX)

## 2. نقاط الـ API — `lib/linkapi.dart`

```dart
abstract class AppLink {
  static const String serverName = "https://api.yourapp.com/api/";

  // ── Auth (تسجيل + OTP) ──────────────────────────────────
  static const String register   = "${serverName}auth/register";
  static const String verifyOtp  = "${serverName}auth/verify-otp";      // POST {otp}
  static const String resendOtp  = "${serverName}auth/resend-otp";       // POST {}

  // ── Forgot Password (3 خطوات) ───────────────────────────
  static const String forgotPassword    = "${serverName}auth/forgot-password";              // POST {email}
  static const String verifyForgotOtp   = "${serverName}auth/forgot-password/verify-otp";  // POST {email, otp}
  static const String resetPassword     = "${serverName}auth/reset-password";               // POST {email, otp, password, password_confirmation}
  static const String resendForgotOtp   = "${serverName}auth/forgot-password/resend-otp";  // POST {email}
}
```

**مفاتيح Body لكل endpoint:**

| Endpoint | Body |
|---|---|
| `register` | `email, first_Name, last_Name, phone, password` |
| `verify-otp` | `otp` |
| `resend-otp` | `{}` (فارغ) |
| `forgot-password` | `email` |
| `forgot-password/verify-otp` | `email, otp` |
| `reset-password` | `email, otp, password, password_confirmation` |
| `forgot-password/resend-otp` | `email` |

## 7. الـ Controllers

### 7.1 — `AuthController` — الجزء المتعلق بـ OTP التسجيل

```dart
// lib/controller/auth/auth_controller.dart
class AuthController extends GetxController {
  final _storage = GetStorage();
  late final AuthData _authData;
  late final RegisterData _registerData;

  final RxBool isLoading = false.obs;
  final Rx<StatusRequest> status = StatusRequest.none.obs;
  final RxString errorMessage = ''.obs;
  Rx<UserModel?> currentUser = Rx<UserModel?>(null);

  @override
  void onInit() {
    super.onInit();
    final crud = Get.find<Crud>();
    _authData = AuthData(crud);
    _registerData = RegisterData(crud);
  }

  // ── Getters: التحقق من وجود جلسة تسجيل معلّقة ──────────
  bool get hasPendingVerification =>
      _storage.read<bool>('pendingVerification') ?? false;

  String get pendingEmail => _storage.read<String>('pendingEmail') ?? '';

  // ── Register: إنشاء الحساب → ينتقل إلى OTP ──────────────
  void register({
    required String identifier,
    required String password,
    required String nameCtrl,
    required String lastNameCtrl,
    required String phoneCtrl,
  }) async {
    isLoading.value = true;
    status.value = StatusRequest.loading;

    final response = await _registerData.postData(
        identifier, password, nameCtrl, lastNameCtrl, phoneCtrl);

    final sr = StatusRequest.values.byName(
        response['statusRequest'] ?? StatusRequest.failure.name);

    if (sr == StatusRequest.success && response['status'] == true) {
      // حفظ بيانات التحقق المعلّق
      final otp = _extractOtp(response);
      _storage.write('pendingVerification', true);
      _storage.write('pendingEmail', identifier);
      if (otp != null) _storage.write('pendingOtp', otp);

      status.value = StatusRequest.success;
      Get.toNamed(AppRoutes.otp);
    } else {
      // معالجة الخطأ...
    }
    isLoading.value = false;
  }

  // ── Verify OTP: التحقق من الحساب الجديد ─────────────────
  void verifyOtp(String otp) async {
    isLoading.value = true;
    status.value = StatusRequest.loading;

    // مقارنة محلية أولاً مع OTP المحفوظ من response التسجيل
    final storedOtp = _storage.read<String>('pendingOtp');
    if (storedOtp != null && storedOtp.isNotEmpty && otp != storedOtp) {
      isLoading.value = false;
      status.value = StatusRequest.failure;
      Get.snackbar('رمز خاطئ', 'الرمز الذي أدخلته غير صحيح',
          snackPosition: SnackPosition.BOTTOM);
      return;
    }

    // إرسال OTP للـ backend للتحقق الرسمي وإصدار الـ token
    final response = await _authData.verifyOtp(otp);
    final sr = StatusRequest.values.byName(
        response['statusRequest'] ?? StatusRequest.failure.name);

    if (sr == StatusRequest.success && response['status'] == true) {
      final auth = AuthResponseModel.fromJson(response);
      if (auth.token != null) _storage.write('token', auth.token);
      if (auth.user != null) currentUser.value = auth.user;

      _clearPendingVerification();
      _storage.write('isLoggedIn', true);
      status.value = StatusRequest.success;
      Get.offAllNamed(AppRoutes.home);
    } else {
      Get.snackbar('رمز خاطئ', response['message']?.toString() ?? 'خطأ',
          snackPosition: SnackPosition.BOTTOM);
    }
    isLoading.value = false;
  }

  // ── Resend OTP ────────────────────────────────────────────
  Future<void> resendOtp() async {
    isLoading.value = true;
    try {
      await _authData.resendOtp();
      _storage.remove('pendingOtp'); // OTP قديم لم يعد صالحاً
      Get.snackbar('تم الإرسال', 'تم إرسال رمز جديد',
          snackPosition: SnackPosition.BOTTOM);
    } catch (_) {
      Get.snackbar('خطأ', 'تعذّر إعادة الإرسال',
          snackPosition: SnackPosition.BOTTOM);
    }
    isLoading.value = false;
  }

  // ── Helpers ───────────────────────────────────────────────
  void _clearPendingVerification() {
    _storage.remove('pendingVerification');
    _storage.remove('pendingEmail');
    _storage.remove('pendingOtp');
  }

  /// استخراج OTP من مختلف أشكال الـ response
  String? _extractOtp(Map<String, dynamic> response) {
    final direct = response['otp'] ?? response['code'] ?? response['verification_code'];
    if (direct != null) return direct.toString();
    final data = response['data'];
    if (data is Map<String, dynamic>) {
      final nested = data['otp'] ?? data['code'] ?? data['verification_code'];
      if (nested != null) return nested.toString();
    }
    return null;
  }
}
```

### 7.2 — `ForgotPasswordController` — 3 خطوات كاملة

```dart
// lib/controller/auth/forgot_password_controller.dart
class ForgotPasswordController extends GetxController {
  final _storage = GetStorage();
  late final ForgotPasswordData _forgotPasswordData;

  final RxBool isLoading = false.obs;
  final Rx<StatusRequest> status = StatusRequest.none.obs;
  final RxString errorMessage = ''.obs;
  final RxString savedEmail = ''.obs;
  final RxString savedOtp = ''.obs;

  // مفاتيح التخزين المحلي للجلسة
  static const _kEmail = 'fp_email';
  static const _kOtp   = 'fp_otp';
  static const _kStep  = 'fp_step'; // 1=OTP, 2=Reset

  @override
  void onInit() {
    super.onInit();
    _forgotPasswordData = ForgotPasswordData(Get.find<Crud>());
    // استعادة جلسة محفوظة عند إعادة فتح التطبيق
    savedEmail.value = _storage.read<String>(_kEmail) ?? '';
    savedOtp.value   = _storage.read<String>(_kOtp)   ?? '';
  }

  // ── Getters للجلسة المعلّقة ──────────────────────────────
  bool get hasPendingSession =>
      (_storage.read<int>(_kStep) ?? 0) > 0;

  int get pendingStep => _storage.read<int>(_kStep) ?? 0;

  void resumePendingSession() {
    if (pendingStep == 1) Get.toNamed(AppRoutes.forgotPasswordOtp);
    else if (pendingStep == 2) Get.toNamed(AppRoutes.resetPassword);
  }

  void discardSession() => _clearSession();

  // ── Step 1: إرسال OTP إلى الإيميل ───────────────────────
  Future<void> sendOtp(String email) async {
    isLoading.value = true;
    status.value = StatusRequest.loading;
    errorMessage.value = '';

    final response = await _forgotPasswordData.sendOtp(email);
    final sr = StatusRequest.values.byName(
        response['statusRequest'] ?? StatusRequest.failure.name);

    if (sr == StatusRequest.success && response['status'] == true) {
      savedEmail.value = email;
      _storage.write(_kEmail, email);
      _storage.write(_kStep, 1);
      _storage.remove(_kOtp);

      status.value = StatusRequest.success;
      Get.toNamed(AppRoutes.forgotPasswordOtp);
    } else {
      _handleError(sr, response['message']);
    }
    isLoading.value = false;
  }

  // ── Step 2: التحقق من OTP ────────────────────────────────
  Future<void> verifyOtp(String otp) async {
    isLoading.value = true;
    status.value = StatusRequest.loading;
    errorMessage.value = '';

    final response = await _forgotPasswordData.verifyOtp(savedEmail.value, otp);
    final sr = StatusRequest.values.byName(
        response['statusRequest'] ?? StatusRequest.failure.name);

    if (sr == StatusRequest.success && response['status'] == true) {
      savedOtp.value = otp;
      _storage.write(_kOtp, otp);
      _storage.write(_kStep, 2);

      status.value = StatusRequest.success;
      Get.toNamed(AppRoutes.resetPassword);
    } else {
      _handleError(sr, response['message']);
    }
    isLoading.value = false;
  }

  // ── Step 3: تعيين كلمة المرور الجديدة ───────────────────
  Future<void> resetPassword(String password, String confirmPassword) async {
    isLoading.value = true;
    status.value = StatusRequest.loading;
    errorMessage.value = '';

    final response = await _forgotPasswordData.resetPassword(
      email: savedEmail.value,
      otp: savedOtp.value,
      password: password,
      passwordConfirmation: confirmPassword,
    );
    final sr = StatusRequest.values.byName(
        response['statusRequest'] ?? StatusRequest.failure.name);

    if (sr == StatusRequest.success && response['status'] == true) {
      _clearSession();
      status.value = StatusRequest.success;
      Get.snackbar('تم بنجاح', 'تم تغيير كلمة المرور، يمكنك تسجيل الدخول الآن',
          snackPosition: SnackPosition.BOTTOM);
      Get.offAllNamed(AppRoutes.login);
    } else {
      _handleError(sr, response['message']);
    }
    isLoading.value = false;
  }

  // ── إعادة إرسال OTP (نسيان كلمة المرور) ─────────────────
  Future<void> resendOtp() async {
    if (savedEmail.value.isEmpty) return;
    isLoading.value = true;
    errorMessage.value = '';

    final response = await _forgotPasswordData.resendOtp(savedEmail.value);
    final sr = StatusRequest.values.byName(
        response['statusRequest'] ?? StatusRequest.failure.name);

    if (sr == StatusRequest.success && response['status'] == true) {
      savedOtp.value = '';
      _storage.remove(_kOtp);
      Get.snackbar('تم الإرسال', 'تم إرسال الرمز مجدداً إلى بريدك الإلكتروني',
          snackPosition: SnackPosition.BOTTOM);
    } else {
      _handleError(sr, response['message']);
    }
    isLoading.value = false;
  }

  // ── Helpers ───────────────────────────────────────────────
  void _clearSession() {
    savedEmail.value = '';
    savedOtp.value   = '';
    _storage.remove(_kEmail);
    _storage.remove(_kOtp);
    _storage.remove(_kStep);
  }

  void _handleError(StatusRequest sr, dynamic message) {
    status.value = sr;
    errorMessage.value = message?.toString() ?? 'حدث خطأ، يرجى المحاولة مجدداً';
    Get.snackbar('خطأ', errorMessage.value, snackPosition: SnackPosition.BOTTOM);
  }
}
```

```dart
// Controller داخلي (غير global) يدير الـ TextEditingControllers والـ FocusNodes
class _OtpFieldRowController extends GetxController {
  late final List<TextEditingController> controllers;
  late final List<FocusNode> focusNodes;
  final int length;

  _OtpFieldRowController(this.length);

  @override
  void onInit() {
    super.onInit();
    controllers = List.generate(length, (_) => TextEditingController());
    focusNodes  = List.generate(length, (_) => FocusNode());
  }

  @override
  void onClose() {
    for (final c in controllers) c.dispose();
    for (final f in focusNodes)  f.dispose();
    super.onClose();
  }

  void onChanged(int i, String val, ValueChanged<String> onCompleted) {
    if (val.length == 1 && i < length - 1) focusNodes[i + 1].requestFocus();
    if (val.isEmpty   && i > 0)            focusNodes[i - 1].requestFocus();
    final code = controllers.map((c) => c.text).join();
    if (code.length == length) onCompleted(code);
  }
}

class OtpFieldRow extends StatelessWidget {
  final int length;
  final ValueChanged<String> onCompleted;
  const OtpFieldRow({super.key, this.length = 6, required this.onCompleted});

      init: _OtpFieldRowController(length),
      global: false, // ← مهم: منع تعارض instances متعددة

        mainAxisAlignment: MainAxisAlignment.center,
```

### 8.2 — نمط الـ Screen Controller الداخلي للـ OTP

> كلتا شاشتَي OTP تستخدمان نفس النمط: controller داخلي (`global: false`) لإدارة العدّاد والـ OTP المُدخَل، مستقل عن الـ main controller.

```dart
class _OtpScreenController extends GetxController {
  final otp     = ''.obs;
  final seconds = 60.obs;
  Timer? _timer;

  @override
  void onInit() { super.onInit(); _startTimer(); }

  @override
  void onClose() { _timer?.cancel(); super.onClose(); }

  void _startTimer() {
    seconds.value = 60;
    _timer?.cancel();
    _timer = Timer.periodic(const Duration(seconds: 1), (t) {
      if (seconds.value == 0) { t.cancel(); return; }
      seconds.value--;
    });
  }

  void updateOtp(String val) => otp.value = val;

  Future<void> resend(/* MainController */ ctrl) async {
    await ctrl.resendOtp();
    otp.value = '';
    _startTimer();
  }
}
```

```dart
class ResetPasswordScreen extends StatelessWidget {
  const ResetPasswordScreen({super.key});

  Widget build(BuildContext context) {
    final ctrl        = Get.find<ForgotPasswordController>();
    final passwordCtrl = TextEditingController();
    final confirmCtrl  = TextEditingController();

    return Scaffold(
      body: Form(

        child: Column(children: [

          // حقل كلمة المرور الجديدة
          TextFormField(
            controller: passwordCtrl,
            obscureText: true,
            validator: /* validPassword: 8+ chars, uppercase, number */,
          ),

          // حقل تأكيد كلمة المرور
          TextFormField(
            controller: confirmCtrl,
            obscureText: true,
            validator: (val) {
              if (val != passwordCtrl.text) return 'كلمتا المرور غير متطابقتين';
              return null;
            },
          ),

          // عداد متطلبات كلمة المرور (live feedback)
          ValueListenableBuilder<TextEditingValue>(
            valueListenable: passwordCtrl,
            builder: (_, value, __) {
              final p = value.text;
              return Column(children: [
                _RequirementRow('8 أحرف على الأقل',      p.length >= 8),
                _RequirementRow('حرف كبير واحد على الأقل', p.contains(RegExp(r'[A-Z]'))),
                _RequirementRow('رقم واحد على الأقل',      p.contains(RegExp(r'[0-9]'))),
              ]);
            },
          ),

          // زر الحفظ
          Obx(() => ElevatedButton(
            onPressed: ctrl.isLoading.value ? null : () {
              if (formKey.currentState?.validate() ?? false) {
                ctrl.resetPassword(
                  passwordCtrl.text.trim(),
                  confirmCtrl.text.trim(),
                );
              }
            },
            child: const Text('حفظ كلمة المرور'),
          )),

          // مؤشر الخطوات (currentStep: 2)
          _StepsIndicator(currentStep: 2),
```
